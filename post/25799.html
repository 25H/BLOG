AMZ 扩展应用html配置接口

 2018年12月08日  笔记 0.32k  1min  25H

自动保存

自动报错目前仅支持 input textarea select 基础标签。

使用方法：

<input type="text" amz-autosave amz-name="背景色" value="#fff" />

其中 amz-autosave 属性代表开启自动保存功能，amz-name="" 属性为保存的键名。

其他接口

注：动作参数不区分大小写。


// 定义通讯接口，其中参数一必须为 'api'
var SendMsg = function (p1, p2, p3, p4, p5) { return parent.SendMsg(p1, p2, p3, p4, p5); };


/**
  * 获取AMZ版本
  *
  * AMZver
  *       Major
  *       Minor
  *       Patch
  *       Tag
  *       Name
  */

// 获取AMZ版本,例子
var AMZver = SendMsg('API', 'AMZver', 'Name');	// AMZ 完整字符串版本 如：3.1.1_Beta


/**
  * 读写配置项
  *
  * Save
  *       配置项名
  *       写入值( \n会被替换为 #换行符# )
  *
  * Read  ( #换行符# 会被替换为\n )
  *       配置项名
  *       默认值
  */

// 读写配置项,例子

// 写配置项
SendMsg('API', 'Save', 'dataJson', JSON.stringify(dataJson));

// 读配置项
var dataJson = SendMsg('API', 'Read', 'dataJson', '');
